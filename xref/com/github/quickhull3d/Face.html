25  /**
26   * Basic triangular face used to form the hull.
27   * <p>
28   * The information stored for each face consists of a planar normal, a planar
29   * offset, and a doubly-linked list of three <a href=HalfEdge>HalfEdges</a>
30   * which surround the face in a counter-clockwise direction.
31   * 
32   * @author John E. Lloyd, Fall 2004
33   */
34  class Face {

68      public static Face create(Vertex[] vtxArray, int[] indices) {
69          Face face = new Face();
70          HalfEdge hePrev = null;
71          for (int i = 0; i < indices.length; i++) {
72              HalfEdge he = new HalfEdge(vtxArray[indices[i]], face);
73              if (hePrev != null) {
74                  he.setPrev(hePrev);
75                  hePrev.setNext(he);
76              } else {
77                  face.he0 = he;
78              }
79              hePrev = he;
80          }
81          face.he0.setPrev(hePrev);
82          hePrev.setNext(face.he0);
83  
84          // compute the normal and offset
85          face.computeNormalAndCentroid();
86          return face;
87      }

89      public static Face createTriangle(Vertex v0, Vertex v1, Vertex v2) {
90          return createTriangle(v0, v1, v2, 0);
91      }

93      /**
94       * Constructs a triangule Face from vertices v0, v1, and v2.
95       * 
96       * @param v0
97       *            first vertex
98       * @param v1
99       *            second vertex
100      * @param v2
101      *            third vertex
102      */
103     public static Face createTriangle(Vertex v0, Vertex v1, Vertex v2, double minArea) {
104         Face face = new Face();
105         HalfEdge he0 = new HalfEdge(v0, face);
106         HalfEdge he1 = new HalfEdge(v1, face);
107         HalfEdge he2 = new HalfEdge(v2, face);
108 
109         he0.prev = he2;
110         he0.next = he1;
111         he1.prev = he0;
112         he1.next = he2;
113         he2.prev = he1;
114         he2.next = he0;
115 
116         face.he0 = he0;
117 
118         // compute the normal and offset
119         face.computeNormalAndCentroid(minArea);
120         return face;
121     }

123     public void computeCentroid(Point3d centroid) {
124         centroid.setZero();
125         HalfEdge he = he0;
126         do {
127             centroid.add(he.head().pnt);
128             he = he.next;
129         } while (he != he0);
130         centroid.scale(1 / (double) numVerts);
131     }

133     public void computeNormal(Vector3d normal) {
134         HalfEdge he1 = he0.next;
135         HalfEdge he2 = he1.next;
136 
137         Point3d p0 = he0.head().pnt;
138         Point3d p2 = he1.head().pnt;
139 
140         double d2x = p2.x - p0.x;
141         double d2y = p2.y - p0.y;
142         double d2z = p2.z - p0.z;
143 
144         normal.setZero();
145 
146         numVerts = 2;
147 
148         while (he2 != he0) {
149             double d1x = d2x;
150             double d1y = d2y;
151             double d1z = d2z;
152 
153             p2 = he2.head().pnt;
154             d2x = p2.x - p0.x;
155             d2y = p2.y - p0.y;
156             d2z = p2.z - p0.z;
157 
158             normal.x += d1y * d2z - d1z * d2y;
159             normal.y += d1z * d2x - d1x * d2z;
160             normal.z += d1x * d2y - d1y * d2x;
161 
162             he1 = he2;
163             he2 = he2.next;
164             numVerts++;
165         }
166         area = normal.norm();
167         normal.scale(1 / area);
168     }

170     public void computeNormal(Vector3d normal, double minArea) {
171         computeNormal(normal);
172 
173         if (area < minArea) {
174             // make the normal more robust by removing
175             // components parallel to the longest edge
176 
177             HalfEdge hedgeMax = null;
178             double lenSqrMax = 0;
179             HalfEdge hedge = he0;
180             do {
181                 double lenSqr = hedge.lengthSquared();
182                 if (lenSqr > lenSqrMax) {
183                     hedgeMax = hedge;
184                     lenSqrMax = lenSqr;
185                 }
186                 hedge = hedge.next;
187             } while (hedge != he0);
188 
189             Point3d p2 = hedgeMax.head().pnt;
190             Point3d p1 = hedgeMax.tail().pnt;
191             double lenMax = Math.sqrt(lenSqrMax);
192             double ux = (p2.x - p1.x) / lenMax;
193             double uy = (p2.y - p1.y) / lenMax;
194             double uz = (p2.z - p1.z) / lenMax;
195             double dot = normal.x * ux + normal.y * uy + normal.z * uz;
196             normal.x -= dot * ux;
197             normal.y -= dot * uy;
198             normal.z -= dot * uz;
199 
200             normal.normalize();
201         }
202     }

204     /**
205      * Computes the distance from a point p to the plane of this face.
206      * 
207      * @param p
208      *            the point
209      * @return distance from the point to the plane
210      */
211     public double distanceToPlane(Point3d p) {
212         return normal.x * p.x + normal.y * p.y + normal.z * p.z - planeOffset;
213     }

215     /**
216      * Finds the half-edge within this face which has tail <code>vt</code> and
217      * head <code>vh</code>.
218      * 
219      * @param vt
220      *            tail point
221      * @param vh
222      *            head point
223      * @return the half-edge, or null if none is found.
224      */
225     public HalfEdge findEdge(Vertex vt, Vertex vh) {
226         HalfEdge he = he0;
227         do {
228             if (he.head() == vh && he.tail() == vt) {
229                 return he;
230             }
231             he = he.next;
232         } while (he != he0);
233         return null;
234     }

240     /**
241      * Gets the i-th half-edge associated with the face.
242      * 
243      * @param i
244      *            the half-edge index, in the range 0-2.
245      * @return the half-edge
246      */
247     public HalfEdge getEdge(int i) {
248         HalfEdge he = he0;
249         while (i > 0) {
250             he = he.next;
251             i--;
252         }
253         while (i < 0) {
254             he = he.prev;
255             i++;
256         }
257         return he;
258     }

282     public String getVertexString() {
283         String s = null;
284         HalfEdge he = he0;
285         do {
286             if (s == null) {
287                 s = "" + he.head().index;
288             } else {
289                 s += " " + he.head().index;
290             }
291             he = he.next;
292         } while (he != he0);
293         return s;

296     public int mergeAdjacentFace(HalfEdge hedgeAdj, Face[] discarded) {
297         Face oppFace = hedgeAdj.oppositeFace();
298         int numDiscarded = 0;
299 
300         discarded[numDiscarded++] = oppFace;
301         oppFace.mark = DELETED;
302 
303         HalfEdge hedgeOpp = hedgeAdj.getOpposite();
304 
305         HalfEdge hedgeAdjPrev = hedgeAdj.prev;
306         HalfEdge hedgeAdjNext = hedgeAdj.next;
307         HalfEdge hedgeOppPrev = hedgeOpp.prev;
308         HalfEdge hedgeOppNext = hedgeOpp.next;
309 
310         while (hedgeAdjPrev.oppositeFace() == oppFace) {
311             hedgeAdjPrev = hedgeAdjPrev.prev;
312             hedgeOppNext = hedgeOppNext.next;
313         }
314 
315         while (hedgeAdjNext.oppositeFace() == oppFace) {
316             hedgeOppPrev = hedgeOppPrev.prev;
317             hedgeAdjNext = hedgeAdjNext.next;
318         }
319 
320         HalfEdge hedge;
321 
322         for (hedge = hedgeOppNext; hedge != hedgeOppPrev.next; hedge = hedge.next) {
323             hedge.face = this;
324         }
325 
326         if (hedgeAdj == he0) {
327             he0 = hedgeAdjNext;
328         }
329 
330         // handle the half edges at the head
331         Face discardedFace;
332 
333         discardedFace = connectHalfEdges(hedgeOppPrev, hedgeAdjNext);
334         if (discardedFace != null) {
335             discarded[numDiscarded++] = discardedFace;
336         }
337 
338         // handle the half edges at the tail
339         discardedFace = connectHalfEdges(hedgeAdjPrev, hedgeOppNext);
340         if (discardedFace != null) {
341             discarded[numDiscarded++] = discardedFace;
342         }
343 
344         computeNormalAndCentroid();
345         checkConsistency();
346 
347         return numDiscarded;
348     }

354     public void triangulate(FaceList newFaces, double minArea) {
355         HalfEdge hedge;
356 
357         if (numVertices() < 4) {
358             return;
359         }
360 
361         Vertex v0 = he0.head();
362         Face prevFace = null;
363 
364         hedge = he0.next;
365         HalfEdge oppPrev = hedge.opposite;
366         Face face0 = null;
367 
368         for (hedge = hedge.next; hedge != he0.prev; hedge = hedge.next) {
369             Face face = createTriangle(v0, hedge.prev.head(), hedge.head(), minArea);
370             face.he0.next.setOpposite(oppPrev);
371             face.he0.prev.setOpposite(hedge.opposite);
372             oppPrev = face.he0;
373             newFaces.add(face);
374             if (face0 == null) {
375                 face0 = face;
376             }
377         }
378         hedge = new HalfEdge(he0.prev.prev.head(), this);
379         hedge.setOpposite(oppPrev);
380 
381         hedge.prev = he0;
382         hedge.prev.next = hedge;
383 
384         hedge.next = he0.prev;
385         hedge.next.prev = hedge;
386 
387         computeNormalAndCentroid(minArea);
388         checkConsistency();
389 
390         for (Face face = face0; face != null; face = face.next) {
391             face.checkConsistency();
392         }
393 
394     }

396     private double areaSquared(HalfEdge hedge0, HalfEdge hedge1) {
397         // return the squared area of the triangle defined
398         // by the half edge hedge0 and the point at the
399         // head of hedge1.
400 
401         Point3d p0 = hedge0.tail().pnt;
402         Point3d p1 = hedge0.head().pnt;
403         Point3d p2 = hedge1.head().pnt;
404 
405         double dx1 = p1.x - p0.x;
406         double dy1 = p1.y - p0.y;
407         double dz1 = p1.z - p0.z;
408 
409         double dx2 = p2.x - p0.x;
410         double dy2 = p2.y - p0.y;
411         double dz2 = p2.z - p0.z;
412 
413         double x = dy1 * dz2 - dz1 * dy2;
414         double y = dz1 * dx2 - dx1 * dz2;
415         double z = dx1 * dy2 - dy1 * dx2;
416 
417         return x * x + y * y + z * z;
418     }

420     private void computeNormalAndCentroid() {
421         computeNormal(normal);
422         computeCentroid(centroid);
423         planeOffset = normal.dot(centroid);
424         int numv = 0;
425         HalfEdge he = he0;
426         do {
427             numv++;
428             he = he.next;
429         } while (he != he0);
430         if (numv != numVerts) {
431             throw new InternalErrorException("face " + getVertexString() + " numVerts=" + numVerts + " should be " + numv);
432         }
433     }
434 
435     private void computeNormalAndCentroid(double minArea) {
436         computeNormal(normal, minArea);
437         computeCentroid(centroid);
438         planeOffset = normal.dot(centroid);
439     }

441     private Face connectHalfEdges(HalfEdge hedgePrev, HalfEdge hedge) {
442         Face discardedFace = null;
443 
444         if (hedgePrev.oppositeFace() == hedge.oppositeFace()) { // then there is
445                                                                 // a redundant
446                                                                 // edge that we
447                                                                 // can get rid
448                                                                 // off
449 
450             Face oppFace = hedge.oppositeFace();
451             HalfEdge hedgeOpp;
452 
453             if (hedgePrev == he0) {
454                 he0 = hedge;
455             }
456             if (oppFace.numVertices() == 3) { // then we can get rid of the
457                                               // opposite face altogether
458                 hedgeOpp = hedge.getOpposite().prev.getOpposite();
459 
460                 oppFace.mark = DELETED;
461                 discardedFace = oppFace;
462             } else {
463                 hedgeOpp = hedge.getOpposite().next;
464 
465                 if (oppFace.he0 == hedgeOpp.prev) {
466                     oppFace.he0 = hedgeOpp;
467                 }
468                 hedgeOpp.prev = hedgeOpp.prev.prev;
469                 hedgeOpp.prev.next = hedgeOpp;
470             }
471             hedge.prev = hedgePrev.prev;
472             hedge.prev.next = hedge;
473 
474             hedge.opposite = hedgeOpp;
475             hedgeOpp.opposite = hedge;
476 
477             // oppFace was modified, so need to recompute
478             oppFace.computeNormalAndCentroid();
479         } else {
480             hedgePrev.next = hedge;
481             hedge.prev = hedgePrev;
482         }
483         return discardedFace;
484     }

486     void checkConsistency() {
487         // do a sanity check on the face
488         HalfEdge hedge = he0;
489         double maxd = 0;
490         int numv = 0;
491 
492         if (numVerts < 3) {
493             throw new InternalErrorException("degenerate face: " + getVertexString());
494         }
495         do {
496             HalfEdge hedgeOpp = hedge.getOpposite();
497             if (hedgeOpp == null) {
498                 throw new InternalErrorException("face " + getVertexString() + ": " + "unreflected half edge " + hedge.getVertexString());
499             } else if (hedgeOpp.getOpposite() != hedge) {
500                 throw new InternalErrorException("face " + getVertexString() + ": " + "opposite half edge " + hedgeOpp.getVertexString() + " has opposite "
501                         + hedgeOpp.getOpposite().getVertexString());
502             }
503             if (hedgeOpp.head() != hedge.tail() || hedge.head() != hedgeOpp.tail()) {
504                 throw new InternalErrorException("face " + getVertexString() + ": " + "half edge " + hedge.getVertexString() + " reflected by " + hedgeOpp.getVertexString());
505             }
506             Face oppFace = hedgeOpp.face;
507             if (oppFace == null) {
508                 throw new InternalErrorException("face " + getVertexString() + ": " + "no face on half edge " + hedgeOpp.getVertexString());
509             } else if (oppFace.mark == DELETED) {
510                 throw new InternalErrorException("face " + getVertexString() + ": " + "opposite face " + oppFace.getVertexString() + " not on hull");
511             }
512             double d = Math.abs(distanceToPlane(hedge.head().pnt));
513             if (d > maxd) {
514                 maxd = d;
515             }
516             numv++;
517             hedge = hedge.next;
518         } while (hedge != he0);
519 
520         if (numv != numVerts) {
521             throw new InternalErrorException("face " + getVertexString() + " numVerts=" + numVerts + " should be " + numv);
522         }
523 
524     }

25  /**
26   * Represents the half-edges that surround each face in a counter-clockwise
27   * direction.
28   * 
29   * @author John E. Lloyd, Fall 2004
30   */
31  class HalfEdge {
32  
33      /**
34       * The vertex associated with the head of this half-edge.
35       */
36      Vertex vertex;
37  
38      /**
39       * Triangular face associated with this half-edge.
40       */
41      Face face;
42  
43      /**
44       * Next half-edge in the triangle.
45       */
46      HalfEdge next;
47  
48      /**
49       * Previous half-edge in the triangle.
50       */
51      HalfEdge prev;
52  
53      /**
54       * Half-edge associated with the opposite triangle adjacent to this edge.
55       */
56      HalfEdge opposite;
57  
58      /**
59       * Constructs a HalfEdge with head vertex <code>v</code> and left-hand
60       * triangular face <code>f</code>.
61       * 
62       * @param v
63       *            head vertex
64       * @param f
65       *            left-hand triangular face
66       */
67      public HalfEdge(Vertex v, Face f) {
68          vertex = v;
69          face = f;
70      }
71  
72      public HalfEdge() {
73      }
74  
75      /**
76       * Sets the value of the next edge adjacent (counter-clockwise) to this one
77       * within the triangle.
78       * 
79       * @param edge
80       *            next adjacent edge
81       */
82      public void setNext(HalfEdge edge) {
83          next = edge;
84      }
85  
86      /**
87       * Gets the value of the next edge adjacent (counter-clockwise) to this one
88       * within the triangle.
89       * 
90       * @return next adjacent edge
91       */
92      public HalfEdge getNext() {
93          return next;
94      }
95  
96      /**
97       * Sets the value of the previous edge adjacent (clockwise) to this one
98       * within the triangle.
99       * 
100      * @param edge
101      *            previous adjacent edge
102      */
103     public void setPrev(HalfEdge edge) {
104         prev = edge;
105     }
106 
107     /**
108      * Gets the value of the previous edge adjacent (clockwise) to this one
109      * within the triangle.
110      * 
111      * @return previous adjacent edge
112      */
113     public HalfEdge getPrev() {
114         return prev;
115     }
116 
117     /**
118      * Returns the triangular face located to the left of this half-edge.
119      * 
120      * @return left-hand triangular face
121      */
122     public Face getFace() {
123         return face;
124     }
125 
126     /**
127      * Returns the half-edge opposite to this half-edge.
128      * 
129      * @return opposite half-edge
130      */
131     public HalfEdge getOpposite() {
132         return opposite;
133     }
134 
135     /**
136      * Sets the half-edge opposite to this half-edge.
137      * 
138      * @param edge
139      *            opposite half-edge
140      */
141     public void setOpposite(HalfEdge edge) {
142         opposite = edge;
143         edge.opposite = this;
144     }
145 
146     /**
147      * Returns the head vertex associated with this half-edge.
148      * 
149      * @return head vertex
150      */
151     public Vertex head() {
152         return vertex;
153     }
154 
155     /**
156      * Returns the tail vertex associated with this half-edge.
157      * 
158      * @return tail vertex
159      */
160     public Vertex tail() {
161         return prev != null ? prev.vertex : null;
162     }
163 
164     /**
165      * Returns the opposite triangular face associated with this half-edge.
166      * 
167      * @return opposite triangular face
168      */
169     public Face oppositeFace() {
170         return opposite != null ? opposite.face : null;
171     }
172 
173     /**
174      * Produces a string identifying this half-edge by the point index values of
175      * its tail and head vertices.
176      * 
177      * @return identifying string
178      */
179     public String getVertexString() {
180         if (tail() != null) {
181             return "" + tail().index + "-" + head().index;
182         } else {
183             return "?-" + head().index;
184         }
185     }
186 
187     /**
188      * Returns the length of this half-edge.
189      * 
190      * @return half-edge length
191      */
192     public double length() {
193         if (tail() != null) {
194             return head().pnt.distance(tail().pnt);
195         } else {
196             return -1;
197         }
198     }
199 
200     /**
201      * Returns the length squared of this half-edge.
202      * 
203      * @return half-edge length squared
204      */
205     public double lengthSquared() {
206         if (tail() != null) {
207             return head().pnt.distanceSquared(tail().pnt);
208         } else {
209             return -1;
210         }
211     }

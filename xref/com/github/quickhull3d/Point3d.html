25  /**
26   * A three-element spatial point. The only difference between a point and a
27   * vector is in the the way it is transformed by an affine transformation. Since
28   * the transform method is not included in this reduced implementation for
29   * QuickHull3D, the difference is purely academic.
30   * 
31   * @author John E. Lloyd, Fall 2004
32   */
33  public class Point3d extends Vector3d {
34  
35      /**
36       * Creates a Point3d and initializes it to zero.
37       */
38      public Point3d() {
39      }
40  
41      /**
42       * Creates a Point3d by copying a vector
43       * 
44       * @param v
45       *            vector to be copied
46       */
47      public Point3d(Vector3d v) {
48          set(v);
49      }
50  
51      /**
52       * Creates a Point3d with the supplied element values.
53       * 
54       * @param x
55       *            first element
56       * @param y
57       *            second element
58       * @param z
59       *            third element
60       */
61      public Point3d(double x, double y, double z) {
62          set(x, y, z);
63      }
64  }
